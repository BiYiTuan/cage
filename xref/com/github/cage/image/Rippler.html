16  package com.github.cage.image;
17  
18  import java.awt.image.BufferedImage;
19  
20  /**
21   * A filter to generate ripple (wave) effected images. Uses a transformed sinus
22   * wave for this. This class is thread safe.
23   * 
24   * @author akiraly
25   * 
26   */
27  public class Rippler {
28  	/**
29  	 * Class to respresent wave tranforming information for an axis.
30  	 */
31  	public static class AxisConfig {
32  		private final double start;
33  
34  		private final double length;
35  
36  		private final double amplitude;
37  
38  		/**
39  		 * Constructor.
40  		 * 
41  		 * @param start
42  		 *            the starting x offset to generate wave values. Should be
43  		 *            between 0 and 2 * {@link Math#PI}.
44  		 * @param length
45  		 *            the length of x to be used to generate wave values. Should
46  		 *            be between 0 and 4 * {@link Math#PI}.
47  		 * @param amplitude
48  		 *            the maximum y value, if it is too big, some important
49  		 *            parts of the image (like the text) can "wave" out on the
50  		 *            top or on the bottom of the image.
51  		 */
52  		public AxisConfig(double start, double length, double amplitude) {
53  			this.start = normalize(start, 2);
54  			this.length = normalize(length, 4);
55  			this.amplitude = amplitude;
56  		}
57  
58  		/**
59  		 * Normalizes parameter to fall into [0, multi * {@link Math#PI}].
60  		 * 
61  		 * @param a
62  		 *            to be normalized
63  		 * @param multi
64  		 *            multiplicator used for end value
65  		 * @return normalized value
66  		 */
67  		protected double normalize(double a, int multi) {
68  			double piMulti = multi * Math.PI;
69  
70  			a = Math.abs(a);
71  			double d = Math.floor(a / piMulti);
72  
73  			return a - d * piMulti;
74  		}
75  
76  		/**
77  		 * @return wave part start value
78  		 */
79  		public double getStart() {
80  			return start;
81  		}
82  
83  		/**
84  		 * @return wave part length
85  		 */
86  		public double getLength() {
87  			return length;
88  		}
89  
90  		/**
91  		 * @return amplitude used to transform the wave part
92  		 */
93  		public double getAmplitude() {
94  			return amplitude;
95  		}
96  	}
97  
98  	private final AxisConfig vertical;
99  
100 	private final AxisConfig horizontal;
101 
102 	/**
103 	 * Constructor.
104 	 * 
105 	 * @param vertical
106 	 *            config to calculate waving deltas from x axis (so to modify y
107 	 *            values), not null
108 	 * @param horizontal
109 	 *            config to calculate waving deltas from y axis (so to modify x
110 	 *            values), not null
111 	 */
112 	public Rippler(AxisConfig vertical, AxisConfig horizontal) {
113 		this.vertical = vertical;
114 		this.horizontal = horizontal;
115 	}
116 
117 	/**
118 	 * Draws a rippled (waved) variant of source into destination.
119 	 * 
120 	 * @param src
121 	 *            to be transformed, not null
122 	 * @param dest
123 	 *            to hold the result, not null
124 	 * @return dest is returned
125 	 */
126 	public BufferedImage filter(BufferedImage src, BufferedImage dest) {
127 		int width = src.getWidth();
128 		int height = src.getHeight();
129 
130 		int[] verticalDelta = calcDeltaArray(vertical, width);
131 
132 		int[] horizontalDelta = calcDeltaArray(horizontal, height);
133 
134 		for (int x = 0; x < width; x++) {
135 			for (int y = 0; y < height; y++) {
136 				int ny = (y + verticalDelta[x] + height) % height;
137 				int nx = (x + horizontalDelta[ny] + width) % width;
138 				dest.setRGB(nx, ny, src.getRGB(x, y));
139 			}
140 		}
141 
142 		return dest;
143 	}
144 
145 	/**
146 	 * Calculates wave delta array.
147 	 * 
148 	 * @param axisConfig
149 	 *            config object to transform the wave, not null
150 	 * @param num
151 	 *            number of points needed, positive
152 	 * @return the calculated num length delta array
153 	 */
154 	protected int[] calcDeltaArray(AxisConfig axisConfig, int num) {
155 		int[] delta = new int[num];
156 
157 		double start = axisConfig.getStart();
158 		double period = axisConfig.getLength() / num;
159 		double amplitude = axisConfig.getAmplitude();
160 
161 		for (int fi = 0; fi < num; fi++) {
162 			delta[fi] = (int) Math.round(amplitude
163 					* Math.sin(start + fi * period));
164 		}
165 
166 		return delta;
167 	}
168 
169 	/**
170 	 * @return vertical config, not null
171 	 */
172 	public AxisConfig getVertical() {
173 		return vertical;
174 	}
175 
176 	/**
177 	 * @return horizontal config, not null
178 	 */
179 	public AxisConfig getHorizontal() {
180 		return horizontal;
181 	}
