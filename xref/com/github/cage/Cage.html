38  /**
39   * Convenient entry class to control captcha generation. This class is thread
40   * safe. Example usage:
41   * 
42   * <p>
43   * 
44   * <pre>
45   * <code>
46   * Cage cage = new {@link GCage}(); // lets make some "G" template captchas
47   * // use new {@link YCage}() for "Y" template or configure it yourself with
48   * // one of the Cage {@link #Cage(Painter, IGenerator, IGenerator, String,
49   * Float, IGenerator, Random) constructors}.
50   * 
51   * // ...
52   * 
53   * String token1 = cage.{@link #getTokenGenerator() getTokenGenerator()}.next();
54   * String token2 = cage.{@link #getTokenGenerator() getTokenGenerator()}.next();
55   * 
56   * cage.{@link #draw(String, OutputStream) draw}(token1, someOutputstream1);
57   * cage.{@link #draw(String, OutputStream) draw}(token2, someOutputstream2);
58   * 
59   * </code>
60   * </pre>
61   * 
62   * </p>
63   * 
64   * @author akiraly
65   * 
66   */

93  	/**
94  	 * Constructor.
95  	 * 
96  	 * @param painter
97  	 *            to be used for painting, can be null
98  	 * @param fonts
99  	 *            generator used to generate fonts for texts, defaults to a
100 	 *            random chooser from some predefined set of fonts, can be null
101 	 * @param foregrounds
102 	 *            generator used to generate colors for texts, defaults to a
103 	 *            random "not-bright-so-it-is-readable-on-white" color
104 	 *            generator, can be null
105 	 * @param format
106 	 *            output format, default "jpeg", can be null
107 	 * @param compressRatio
108 	 *            a number in [0f, 1f] interval if compression should be used
109 	 *            with the output format. The format must support compression
110 	 *            (like jpeg and png). If null no compression is done.
111 	 * @param tokenGenerator
112 	 *            a custom String token generator, can be null. If null is
113 	 *            passed a default is created. It is not used by Cage it is only
114 	 *            stored for convenience. Can be retrieved by
115 	 *            {@link #getTokenGenerator()}.
116 	 * @param rnd
117 	 *            random generator to be used, can be null
118 	 */
119 	public Cage(Painter painter, IGenerator<Font> fonts,
120 			IGenerator<Color> foregrounds, String format, Float compressRatio,
121 			IGenerator<String> tokenGenerator, Random rnd) {
122 		if (rnd == null) {
123 			rnd = new Random();
124 		}
125 		this.painter = painter != null ? painter : new Painter(rnd);
126 		int defFontHeight = this.painter.getHeight() / 2;
127 		this.fonts = fonts != null ? fonts : new ObjectRoulette<Font>(rnd, //
128 				new Font(Font.SANS_SERIF, Font.PLAIN, defFontHeight), //
129 				// new Font(Font.SANS_SERIF, Font.ITALIC, defFontHeight),//
130 				new Font(Font.SERIF, Font.PLAIN, defFontHeight), //
131 				// new Font(Font.SERIF, Font.ITALIC, defFontHeight), //
132 				new Font(Font.MONOSPACED, Font.BOLD, defFontHeight)); //
133 		// new Font(Font.MONOSPACED, Font.ITALIC, defFontHeight));
134 		this.foregrounds = foregrounds != null ? foregrounds
135 				: new RgbColorGenerator(rnd);
136 		this.format = format != null ? format : DEFAULT_FORMAT;
137 		this.compressRatio = compressRatio;
138 		this.tokenGenerator = tokenGenerator != null ? tokenGenerator
139 				: new RandomTokenGenerator(rnd);
140 	}

142 	/**
143 	 * Generate an image and serialize it to the output. This method can call
144 	 * {@link OutputStream#close()} on the supplied output stream.
145 	 * 
146 	 * @param text
147 	 *            to be drawn on the image
148 	 * @param ostream
149 	 *            captcha image is serialized to this
150 	 * @throws IOException
151 	 *             if IO error occurs.
152 	 */
153 	public void draw(String text, OutputStream ostream) throws IOException {
154 		BufferedImage img = drawImage(text);
155 		serialize(img, ostream);
156 	}
157 
158 	/**
159 	 * Generate an image and return it in a byte array.
160 	 * 
161 	 * @param text
162 	 *            to be drawn on the image
163 	 * @return byte array holding the serialized generated image
164 	 */
165 	public byte[] draw(String text) {
166 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
167 		try {
168 			try {
169 				draw(text, baos);
170 			} finally {
171 				baos.close();
172 			}
173 		} catch (IOException e) {
174 			// IO errors should not happen we were writing to memory
175 			throw new RuntimeException(
176 					"IO error while writing captcha image to memory.", e);
177 		}
178 
179 		return baos.toByteArray();
180 	}

182 	/**
183 	 * Generates a captcha image.
184 	 * 
185 	 * @param text
186 	 *            to be drawn
187 	 * @return generated image
188 	 */
189 	public BufferedImage drawImage(String text) {
190 		Font font = fonts.next();
191 		Color fground = foregrounds.next();
192 		return painter.draw(font, fground, text);
193 	}

195 	/**
196 	 * Serializes an image to an {@link OutputStream}. This method can call
197 	 * {@link OutputStream#close()} on the supplied output stream.
198 	 * 
199 	 * @param img
200 	 *            to be serialized
201 	 * @param ostream
202 	 *            to be written to
203 	 * @throws IOException
204 	 *             if IO error occurs.
205 	 */
206 	protected void serialize(BufferedImage img, OutputStream ostream)
207 			throws IOException {
208 		ImageTypeSpecifier type = ImageTypeSpecifier
209 				.createFromRenderedImage(img);
210 		Iterator<ImageWriter> iwi = ImageIO.getImageWriters(type, this.format);
211 		if (iwi == null || !iwi.hasNext()) {
212 			throw new IllegalStateException(
213 					"No image writer found for format = " + this.format);
214 		}
215 		ImageWriter iw = iwi.next();
216 		try {
217 			ImageWriteParam iwp;
218 			if (compressRatio != null) {
219 				iwp = iw.getDefaultWriteParam();
220 				iwp.setCompressionMode(ImageWriteParam.MODE_EXPLICIT);
221 				iwp.setCompressionQuality(compressRatio);
222 			} else {
223 				iwp = null;
224 			}
225 			ImageOutputStream ios = ImageIO.createImageOutputStream(ostream);
226 			try {
227 				iw.setOutput(ios);
228 				iw.write(null, new IIOImage(img, null, null), iwp);
229 			} finally {
230 				ios.close();
231 			}
232 		} finally {
233 			iw.dispose();
234 		}
235 	}
